VENDAS:

Lucro Total = SUM('dw fato_vendas'[lucro])

Margem % = DIVIDE([Lucro Total], [Receita Total])

Produto Venda = SUM('dw fato_vendas'[quantidade_itens])

Qtd Pedidos = DISTINCTCOUNT('dw fato_vendas'[id_pedido_origem])

Receita Total = SUM('dw fato_vendas'[valor_liquido])

Ticket Médio = DIVIDE([Receita Total], [Qtd Pedidos])


CLIENTES:

Clientes Ativos = DISTINCTCOUNT('dw fato_vendas'[sk_cliente])
Receita Por Cliente = DIVIDE([Receita Total], [Clientes Ativos])
Média Avaliação = AVERAGE('dw fato_vendas'[avaliacao])
Qtd Clientes = DISTINCTCOUNT('dw dim_cliente'[sk_cliente])

Localização:

Prazo Medio = AVERAGE('dw fato_vendas'[prazo_entrega_dias])

% Entregas Atrasadas = 
DIVIDE(
    CALCULATE(
        COUNTROWS('dw fato_vendas'),
        'dw dim_entrega'[entrega_atrasada] = "Sim"
    ),
    COUNTROWS('dw fato_vendas')
)

Frete Medio = AVERAGE('dw fato_vendas'[valor_frete])

Qtd Atrasos = CALCULATE(COUNTROWS('dw fato_vendas'),'dw dim_entrega'[entrega_atrasada] = "Sim")
